function createGenGrowthKeywordSheet() {

var ss = SpreadsheetApp.create('GenGrowth 关键词研究主表 (科学审计版)');

var master = ss.getActiveSheet();

master.setName('关键词主表');

  

var headers = [

'关键词',

'来源',

'月搜索量',

'KD',

'CPC($)',

'Trends比值',

'SERP平均DR',

'策略覆盖',

'意图词',

'DR过滤',

'分桶',

'内容状态',

'发布URL',

'备注'

];

  

master.getRange(1, 1, 1, headers.length).setValues([headers])

.setBackground('#1a237e').setFontColor('#ffffff').setFontWeight('bold').setFontSize(11);

master.setFrozenRows(1);

  

master.getRange('N1').setValue('我方DR基准: 10');

master.getRange('N1').setFontWeight('bold').setFontColor('#1a237e');

  

var sourceList = ['竞品映射', '内容缺口', '种子词拓展', '社区挖掘', '趋势词', 'Social信号'];

master.getRange('B2:B500').setDataValidation(SpreadsheetApp.newDataValidation().requireValueInList(sourceList, true).build());

master.getRange('L2:L500').setDataValidation(SpreadsheetApp.newDataValidation().requireValueInList(['未开始', '写作中', '已发布', '暂缓'], true).build());

master.getRange('H2:H500').insertCheckboxes();

  

var intentF = '=IF(A2="","",IF(OR(ISNUMBER(SEARCH("how to",A2)),ISNUMBER(SEARCH("fix",A2)),ISNUMBER(SEARCH("best ",A2)),ISNUMBER(SEARCH(" vs",A2)),ISNUMBER(SEARCH("alternative",A2)),ISNUMBER(SEARCH("pricing",A2))),"YES","no"))';

master.getRange('I2:I500').setFormula(intentF);

  

var drF = '=IF(G2="", "✅通过", IF((G2 - 10) > 35, "❌权重墙", "✅通过"))';

master.getRange('J2:J500').setFormula(drF);

  

var bucketF =

'=IF(A2="", "", ' +

'IF(J2="❌权重墙", "🎯战略攻坚", ' +

'IF(H2=TRUE, "⚡快速胜利★", ' +

'IF(OR(B2="趋势词", B2="Social信号", AND(ISNUMBER(F2), F2>1.2)), "🚀趋势词", ' +

'IF(B2="社区挖掘", "💎社区高潜", ' +

'IF(AND(ISNUMBER(D2), D2<20, J2="✅通过", OR(ISNUMBER(C2) && C2>=100, I2="YES")), "⚡快速胜利", ' +

'IF(AND(ISNUMBER(D2), D2<50), "🎯战略词", "📌长尾词")))))))';

master.getRange('K2:K500').setFormula(bucketF);

  

var bucketRange = master.getRange('K2:K500');

var rules = [

{text: '🚀趋势词', bg: '#c8e6c9'},

{text: '💎社区高潜', bg: '#e1f5fe'},

{text: '⚡快速胜利', bg: '#fff9c4'},

{text: '🎯战略攻坚', bg: '#f5f5f5'},

{text: '❌权重墙', bg: '#f5f5f5'}

];

  

var conditionalRules = master.getConditionalFormatRules();

rules.forEach(function(r) {

conditionalRules.push(

SpreadsheetApp.newConditionalFormatRule()

.whenTextContains(r.text)

.setBackground(r.bg)

.setRanges([bucketRange])

.build()

);

});

master.setConditionalFormatRules(conditionalRules);

  

master.setColumnWidth(1, 250);

master.setColumnWidth(2, 100);

master.setColumnWidth(7, 100);

master.setColumnWidth(11, 140);

  

Logger.log('✅ 模板创建成功！');

}
